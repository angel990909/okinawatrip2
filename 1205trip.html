        function switchDay(day) {

            // Update Tab UI

            document.querySelectorAll('.tab-btn').forEach(btn => btn.classList.remove('active'));

            document.getElementById(`tab-${day}`).classList.add('active');

           

            // Render Content

            renderDay(day);

           

            // Scroll top slightly

            window.scrollTo({ top: 0, behavior: 'smooth' });

        }
