            container.innerHTML = '';

                const navBtn = item.nav ? `

        function getBorderColor(tag) {

            if(tag === 'tag-flight') return 'border-l-blue-300';

            if(tag === 'tag-drive') return 'border-l-orange-300';

            if(tag === 'tag-eat') return 'border-l-pink-400';

            if(tag === 'tag-play') return 'border-l-green-300';

            if(tag === 'tag-shop') return 'border-l-purple-300';

            return 'border-l-slate-300';

        }

 

        function switchDay(day) {

            // Update Buttons

            document.querySelectorAll('.nav-btn').forEach(btn => btn.classList.remove('active'));

            document.getElementById(`btn-${day}`).classList.add('active');

            

            // Render

            renderDay(day);

            

            // Scroll Top

            window.scrollTo({top: 0, behavior: 'smooth'});

        }
